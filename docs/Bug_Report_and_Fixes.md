    try:
        from model.resunet_a import build_resunet_a

        # Fallback for different directory structures
        import sys, os
        current_dir = os.path.dirname(__file__)
        model_dir = os.path.join(current_dir, 'model')

            sys.path.insert(0, model_dir)
